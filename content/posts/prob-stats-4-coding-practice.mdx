### 과제 1-1. 평균 신뢰구간 (scipy 없이)

표본 데이터가 주어졌을 때, **scipy를 사용하지 않고** 95% 신뢰구간을 계산하는 함수를 구현하세요.

```python
    """

print(f"95% CI: ({lower:.2f}, {upper:.2f})")
```

```python
def confidence_interval(data: np.ndarray, confidence: float = 0.95) -> tuple[float, float]:
    n = len(data)
    mean = np.mean(data)
    se = np.std(data, ddof=1) / np.sqrt(n)  # ddof=1: 불편 표준편차
    df = n - 1
    alpha = 1 - confidence
    t_critical = stats.t.ppf(1 - alpha / 2, df=df)  # 양측 임계값
    margin = t_critical * se
    return mean - margin, mean + margin

np.random.seed(42)
data = np.random.normal(loc=170, scale=10, size=30)
lower, upper = confidence_interval(data)
print(f"표본 평균: {np.mean(data):.2f}")
print(f"95% CI: ({lower:.2f}, {upper:.2f})")
```

**핵심 포인트**:
- `ddof=1`로 불편 표준편차 사용 (표본이므로 $n-1$로 나눔)
- `stats.t.ppf(1 - α/2, df)`: 양측 검정의 임계값 ($t^*$)
- 신뢰수준 95% → $\alpha=0.05$ → 각 꼬리에 2.5%

</details>

"95% CI를 100번 구하면 약 95번은 모평균을 포함한다"를 시뮬레이션으로 확인하세요.

```python
) -> float:

    주어진 횟수만큼 CI를 생성하고, 실제 모평균을 포함한 비율 반환
    """
```

```python
def simulate_ci_coverage(
    true_mean: float = 0.0,
    true_std: float = 1.0,
    n: int = 30,
    n_simulations: int = 200,
    confidence: float = 0.95
) -> float:
    included = 0
    for _ in range(n_simulations):
        sample = np.random.normal(true_mean, true_std, n)
        lower, upper = confidence_interval(sample, confidence)
        if lower <= true_mean <= upper:
            included += 1
    return included / n_simulations

np.random.seed(0)
coverage = simulate_ci_coverage(n_simulations=1000)
print(f"실제 포함 비율 (1000회): {coverage:.1%}")
# 출력 예: 실제 포함 비율 (1000회): 94.9%
```

**통찰**: 횟수를 늘릴수록 95%에 수렴합니다. "CI가 모수를 포함할 확률"이 아니라 "절차의 장기 성공률"임을 체감할 수 있습니다.

```python
def one_sample_ttest(
    data: np.ndarray,
    mu0: float,
    alternative: str = "two-sided"
) -> tuple[float, float]:
    n = len(data)
    mean = np.mean(data)
    se = np.std(data, ddof=1) / np.sqrt(n)
    t_stat = (mean - mu0) / se
    df = n - 1

    if alternative == "two-sided":
        p_value = 2 * stats.t.sf(abs(t_stat), df)
    elif alternative == "greater":
        p_value = stats.t.sf(t_stat, df)
    elif alternative == "less":
        p_value = stats.t.cdf(t_stat, df)
    else:
        raise ValueError("alternative must be 'two-sided', 'greater', or 'less'")

    return t_stat, p_value

np.random.seed(7)
battery_life = np.random.normal(490, 20, 25)
t_stat, p_val = one_sample_ttest(battery_life, mu0=500)
print(f"t 통계량: {t_stat:.4f}, p값: {p_val:.4f}")
# 결론: p < 0.05면 "광고와 다르다" 판정
```

**핵심**:
- `stats.t.sf(t, df)` = P(T > t): 우측 꼬리 확률 (생존 함수)
- 양측: 절댓값에 2를 곱해 양쪽 꼬리 합산
- 단측(우/좌): 한쪽 꼬리만 사용

```python
print(f"t={t_stat:.4f}, p={p_val:.4f}, df={df:.1f}")
```

```python
def two_sample_ttest_welch(
    group1: np.ndarray,
    group2: np.ndarray,
    alternative: str = "two-sided"
) -> tuple[float, float, float]:
    n1, n2 = len(group1), len(group2)
    mean1, mean2 = np.mean(group1), np.mean(group2)
    var1 = np.var(group1, ddof=1)
    var2 = np.var(group2, ddof=1)

    se = np.sqrt(var1 / n1 + var2 / n2)
    t_stat = (mean1 - mean2) / se

    # Welch-Satterthwaite 자유도 근사
    df = (var1/n1 + var2/n2)**2 / (
        (var1/n1)**2 / (n1 - 1) + (var2/n2)**2 / (n2 - 1)
    )

    if alternative == "two-sided":
        p_value = 2 * stats.t.sf(abs(t_stat), df)
    elif alternative == "greater":
        p_value = stats.t.sf(t_stat, df)
    else:
        p_value = stats.t.cdf(t_stat, df)

    return t_stat, p_value, df

# scipy 검증
t_ref, p_ref = stats.ttest_ind(model_a, model_b, equal_var=False)
print(f"scipy: t={t_ref:.4f}, p={p_ref:.4f}")
```

## Part 4. 대응표본 t검정

### 과제 4-1. 전후 비교 구현

동일한 사용자에게 UI 변경 전후 작업 완료 시간을 측정했습니다. 대응표본 t검정을 구현하세요.

```python
def paired_ttest(
    before: np.ndarray,
    after: np.ndarray,
    alternative: str = "two-sided"
) -> tuple[float, float]:
    """
    대응표본 t검정: 차이 d = after - before
    Returns: (t_statistic, p_value)
    """
    pass

np.random.seed(10)
n_users = 30
before = np.random.normal(120, 20, n_users)   # 초 단위 완료 시간
improvement = np.random.normal(10, 5, n_users) # 개선 효과
after = before - improvement + np.random.normal(0, 3, n_users)

t_stat, p_val = paired_ttest(before, after, alternative="greater")
print(f"t={t_stat:.4f}, p={p_val:.4f}")
print(f"결론: {'UI 개선이 유의미하게 시간 단축' if p_val < 0.05 else '유의한 개선 없음'}")
```

```python
def paired_ttest(
    before: np.ndarray,
    after: np.ndarray,
    alternative: str = "two-sided"
) -> tuple[float, float]:
    d = after - before  # 차이 계산
    n = len(d)
    mean_d = np.mean(d)
    se_d = np.std(d, ddof=1) / np.sqrt(n)
    t_stat = mean_d / se_d
    df = n - 1

    if alternative == "two-sided":
        p_value = 2 * stats.t.sf(abs(t_stat), df)
    elif alternative == "greater":
        p_value = stats.t.sf(t_stat, df)
    else:
        p_value = stats.t.cdf(t_stat, df)

    return t_stat, p_value

# before > after이면 시간 단축 → d = after - before < 0 → t < 0 → "less"
# "greater" 검정: d가 양수인지 (after > before인지) 검정
# 시간이 줄었는지 확인하려면 alternative="less"가 직관적
t_stat, p_val = paired_ttest(before, after, alternative="less")
print(f"시간 단축 검정: t={t_stat:.4f}, p={p_val:.4f}")

# scipy 검증
t_ref, p_ref = stats.ttest_rel(before, after, alternative="less")
print(f"scipy: t={t_ref:.4f}, p={p_ref:.4f}")
```

**핵심**: 대응표본 t검정은 결국 **차이 d의 단일표본 t검정**입니다. `d = after - before`로 개인 간 변동을 흡수해 더 민감한 검정이 됩니다.

</details>

```python
    }
    """
```

```python
def simulate_early_stopping(
    n_max: int = 1000,
    check_every: int = 50,
    alpha: float = 0.05,
    n_experiments: int = 500
) -> dict:
    true_p = 0.1  # 실제로는 차이 없음
    early_stop_errors = 0
    proper_errors = 0

    for _ in range(n_experiments):
        # 데이터 사전 생성
        a_data = np.random.binomial(1, true_p, n_max)
        b_data = np.random.binomial(1, true_p, n_max)

        # 조기 종료: 중간중간 매번 검정
        peeked_significant = False
        for n in range(check_every, n_max + 1, check_every):
            a_sub = a_data[:n]
            b_sub = b_data[:n]
            p_a, p_b = np.mean(a_sub), np.mean(b_sub)
            p_pool = (np.sum(a_sub) + np.sum(b_sub)) / (2 * n)
            se = np.sqrt(p_pool * (1 - p_pool) * (2 / n))
            if se == 0:
                continue
            z = (p_a - p_b) / se
            p_val = 2 * stats.norm.sf(abs(z))
            if p_val < alpha:
                peeked_significant = True
                break

        if peeked_significant:
            early_stop_errors += 1

        # 정상 종료: 끝에서만 한 번 검정
        p_a_final, p_b_final = np.mean(a_data), np.mean(b_data)
        p_pool_final = (p_a_final + p_b_final) / 2
        se_final = np.sqrt(p_pool_final * (1 - p_pool_final) * (2 / n_max))
        if se_final > 0:
            z_final = (p_a_final - p_b_final) / se_final
            p_val_final = 2 * stats.norm.sf(abs(z_final))
            if p_val_final < alpha:
                proper_errors += 1

    return {
        "early_stop_error_rate": early_stop_errors / n_experiments,
        "proper_error_rate": proper_errors / n_experiments
    }

np.random.seed(0)
results = simulate_early_stopping()
print(f"조기 종료 1종 오류율: {results['early_stop_error_rate']:.1%}  (명목 α=5%)")
print(f"정상 종료 1종 오류율: {results['proper_error_rate']:.1%}  (명목 α=5%)")
# 출력 예:
# 조기 종료 1종 오류율: 26.4%  (명목 α=5%)
# 정상 종료 1종 오류율:  5.2%  (명목 α=5%)
```

**결론**: 조기 종료를 허용하면 명목 $\alpha=5\%$임에도 실제 오류율이 20~30%까지 치솟습니다. Sequential testing 방법(예: O'Brien-Fleming, SPRT)을 사용하면 중간 확인과 오류율 통제를 동시에 달성할 수 있습니다.

## 보너스: 전체 파이프라인

```python
def ab_test_pipeline(
    control_conversions: int,
    control_total: int,
    treatment_conversions: int,
    treatment_total: int,
    alpha: float = 0.05
) -> None:
    """A/B 테스트 결과 보고서 출력"""
    p_c = control_conversions / control_total
    p_t = treatment_conversions / treatment_total
    p_pool = (control_conversions + treatment_conversions) / (control_total + treatment_total)
    se = np.sqrt(p_pool * (1 - p_pool) * (1/control_total + 1/treatment_total))
    z = (p_t - p_c) / se
    p_val = 2 * stats.norm.sf(abs(z))

    ci_diff = (p_t - p_c) + np.array([-1, 1]) * stats.norm.ppf(1 - alpha/2) * se

    print("=" * 45)
    print("A/B 테스트 결과 보고서")
    print("=" * 45)
    print(f"Control  전환율: {p_c:.1%}  (n={control_total:,})")
    print(f"Treatment 전환율: {p_t:.1%}  (n={treatment_total:,})")
    print(f"차이: {p_t - p_c:+.2%}")
    print(f"95% CI (차이): [{ci_diff[0]:.2%}, {ci_diff[1]:.2%}]")
    print(f"z 통계량: {z:.3f}  |  p값: {p_val:.4f}")
    print("-" * 45)
    if p_val < alpha:
        print(f"결론: p={p_val:.4f} < α={alpha} → 귀무가설 기각")
        print("      Treatment가 통계적으로 유의한 차이를 보입니다.")
    else:
        print(f"결론: p={p_val:.4f} ≥ α={alpha} → 귀무가설 기각 실패")
        print("      유의한 차이를 발견하지 못했습니다.")
    print("=" * 45)

# 예시 실행
ab_test_pipeline(
    control_conversions=500,
    control_total=10000,
    treatment_conversions=560,
    treatment_total=10000
)
```
